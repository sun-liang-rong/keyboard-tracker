新数据库 Schema 设计

     1. DailyStats 表（每日统计）

     interface DailyStats {
       id: string                    // UUID 或日期作为主键
       date: string                  // YYYY-MM-DD
       total_keystrokes: number      // 总敲击次数
       active_minutes: number        // 活跃分钟数
       peak_hour: number             // 最活跃小时 (0-23)
       created_at: string            // ISO 时间戳
       updated_at: string            // ISO 时间戳

       // 扩展字段：保留高级统计
       category_letter: number       // 字母键
       category_number: number       // 数字键
       category_function: number     // 功能键
       category_control: number      // 控制键
       category_symbol: number       // 符号键
       category_modifier: number     // 修饰键
       category_other: number        // 其他

       // 组合键统计（可选：单独建表）
       combo_copy: number
       combo_paste: number
       combo_cut: number
       combo_select_all: number
       combo_undo: number
       combo_redo: number
       combo_save: number
       // ... 其他组合键
     }

     2. TimeSlotStats 表（时间段统计）

     interface TimeSlotStats {
       id: string
       date: string                  // YYYY-MM-DD
       hour: number                  // 0-23
       keystrokes: number            // 本小时敲击次数
       active_minutes: number        // 本小时活跃分钟数
     }

     3. TopKeys 表（高频按键统计）

     interface TopKeyStats {
       id: string
       date: string                  // YYYY-MM-DD
       key_name: string              // 按键名
       key_count: number             // 按键次数
       key_category: string          // 分类
     }

     4. UnlockedTitles 表（已解锁称号）

     interface UnlockedTitle {
       id: string
       title_id: string              // 称号ID
       unlocked_date: string         // 解锁日期
     }
